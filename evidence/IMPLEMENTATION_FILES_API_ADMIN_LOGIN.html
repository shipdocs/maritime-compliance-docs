Security Event Logging Implementation

// Helper function to log failed login attempts as security events
function logFailedLoginAttempt(email, ipAddress, userAgent, reason) {
  // Use setTimeout to make this non-blocking
  setTimeout(async () => {
    try {
      const { error } = await supabase
        .from('security_events')
        .insert({
          event_type: 'login_failed',
          user_email: email,
          ip_address: ipAddress,
          user_agent: userAgent,
          details: {
            reason: reason,
            endpoint: '/api/auth/admin-login',
            timestamp: new Date().toISOString()
          },
          severity: 'medium'
        });

      if (error) {
        console.error('Failed to log security event:', error);
      }
    } catch (err) {
      console.error('Error in security event logging:', err);
    }
  }, 0);
}

async function validateAdminLoginRequest(req) {

  const validation = validateObject(body, adminLoginSchema);

  return sanitizeAuthInput(validation.data);

Multi-Factor Authentication Integration

// MFA verification for admin accounts
async function verifyAdminMFA(userId, mfaCode, ipAddress) {
  if (!mfaCode) {
    return { success: false, error: 'MFA code required for admin accounts' };
  }
  
  try {
    // Use MFA service to verify code
    const verification = await mfaService.verifyTOTP(userId, mfaCode, ipAddress);
    
    if (!verification.success) {
      // Log MFA failure
      await logAuthSecurityEvent('mfa_verification_failed', {
        userId: userId,
        ipAddress: ipAddress,
        endpoint: '/api/auth/admin-login',
        reason: verification.error
      });
      
      return verification;
    }
    
    return verification;
  } catch (error) {
    console.error('MFA verification error:', error);
    return { success: false, error: 'MFA verification failed' };
  }
}

Database Authentication Query

// Secure user authentication query
async function authenticateAdmin(email, password) {
  try {
    // Query user with role restriction
    const { data: user, error } = await supabase
      .from('users')
      .select('id, email, password_hash, role, is_active, mfa_enabled, failed_login_attempts, locked_until')
      .eq('email', email)
      .eq('role', 'admin') // Only admin users
      .eq('is_active', true) // Only active accounts
      .single();

    if (error) {
      if (error.code === 'PGRST116') {
        return { success: false, error: 'Invalid credentials' };
      }
      throw createDatabaseError('Authentication query failed', error);
    }

    if (!user) {
      return { success: false, error: 'Invalid credentials' };
    }

    // Check account lockout
    if (user.locked_until && new Date(user.locked_until) > new Date()) {
      const lockoutMessage = formatLockoutMessage(user.locked_until);
      return { success: false, error: lockoutMessage, locked: true };
    }

    // Verify password using bcrypt
    const passwordValid = await bcrypt.compare(password, user.password_hash);
    
    if (!passwordValid) {
      // Increment failed attempts
      await incrementFailedAttempts(user.id);
      return { success: false, error: 'Invalid credentials' };
    }

    return { success: true, user };
  } catch (error) {
    console.error('Admin authentication error:', error);
    throw error;
  }
}

Token Generation and Response

// Generate secure token pair for admin session
async function createAdminSession(user, req) {
  try {
    // Generate JWT token pair with enhanced security
    const tokenResult = await generateTokenPair(user, req);
    
    if (!tokenResult.success) {
      throw createError('Failed to generate authentication tokens', 500);
    }

    // Reset failed login attempts on successful login
    await resetFailedAttempts(user.id);

    // Log successful admin login
    await logAuthSecurityEvent('admin_login_success', {
      userId: user.id,
      userEmail: user.email,
      ipAddress: req.headers['x-forwarded-for'] || req.connection.remoteAddress,
      userAgent: req.headers['user-agent'],
      endpoint: '/api/auth/admin-login'
    });

    // Create success response
    return createAuthSuccessResponse({
      user: {
        id: user.id,
        email: user.email,
        role: user.role,
        firstName: user.first_name,
        lastName: user.last_name
      },
      tokens: {
        access_token: tokenResult.accessToken,
        refresh_token: tokenResult.refreshToken,
        expires_in: tokenResult.expiresIn,
        token_type: tokenResult.tokenType

Main Handler Implementation

// Main admin login handler with comprehensive security
const handler = createAPIHandler(async (req, res) => {
  // Only allow POST requests
  if (req.method !== 'POST') {
    throw createError('Method not allowed', 405);
  }

  try {
    // Apply rate limiting
    await adminLoginRateLimit(req, res);
    
    // Validate and sanitize request
    const { email, password, mfaCode } = await validateAdminLoginRequest(req);
    
    // Authenticate user
    const authResult = await authenticateAdmin(email, password);
    
    if (!authResult.success) {
      // Log failed attempt
      logFailedLoginAttempt(
        email,
        req.headers['x-forwarded-for'] || req.connection.remoteAddress,
        req.headers['user-agent'],
        authResult.error
      );
      
      throw createAuthError(authResult.error, authResult.locked ? 423 : 401);
    }

    const { user } = authResult;

    // Verify MFA if enabled
    if (user.mfa_enabled) {
      const mfaResult = await verifyAdminMFA(
        user.id, 
        mfaCode, 
        req.headers['x-forwarded-for'] || req.connection.remoteAddress
      );
      
      if (!mfaResult.success) {
        throw createAuthError(mfaResult.error, 401);
      }
    }

    // Create admin session
    const sessionResponse = await createAdminSession(user, req);
    
    return res.status(200).json(sessionResponse);
    
  } catch (error) {
    // Handle various error types appropriately
    if (error.statusCode) {
      return res.status(error.statusCode).json({ error: error.message });
    }
    
    console.error('Unexpected admin login error:', error);
    return res.status(500).json({ error: 'Internal server error' });
  }
});
